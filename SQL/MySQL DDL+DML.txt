CREATE DATABASE timple;
USE timple;
CREATE TABLE member (
	memberId INTEGER PRIMARY KEY AUTO_INCREMENT,
	name NVARCHAR(50) NOT NULL,
	email NVARCHAR(50) NOT NULL,
	tel NVARCHAR(20) NULL,
	pw NVARCHAR(41) NULL,
	inform NVARCHAR(10) DEFAULT 'yes' NULL,
	alertCount INTEGER DEFAULT 0 NULL,
	profile NVARCHAR(50) NULL,
	socialType CHAR(2) NULL
);

CREATE TABLE notification (
	NotificationId INTEGER PRIMARY KEY AUTO_INCREMENT,
	memseq INTEGER NOT NULL,
	content NVARCHAR(1000) NOT NULL,
	confirm NVARCHAR(10) NOT NULL,
	FOREIGN KEY(memseq) REFERENCES member(memberId) 
);

CREATE TABLE question (
	questionid INTEGER PRIMARY KEY AUTO_INCREMENT,
	content NVARCHAR(50) NULL
);

CREATE TABLE memquestion (
	memquestionId INTEGER PRIMARY KEY AUTO_INCREMENT,
	memseq INTEGER NOT NULL,
	questionSeq INTEGER NOT NULL,
	quereply NVARCHAR(50) NULL,
	FOREIGN KEY(memseq) REFERENCES member(memberId),
	FOREIGN KEY(questionSeq) REFERENCES question(questionid)
);

CREATE TABLE memo (
	memoid INTEGER PRIMARY KEY AUTO_INCREMENT,
	memseq INTEGER NOT NULL,
	content NVARCHAR(100) NULL,
	FOREIGN KEY(memseq) REFERENCES member(memberId)
);

CREATE TABLE friend (
	friendid INTEGER PRIMARY KEY AUTO_INCREMENT,
	memseq INTEGER NOT NULL,
	friendseq INTEGER NOT NULL,
	FOREIGN KEY(memseq) REFERENCES member(memberId),
	FOREIGN KEY(friendseq) REFERENCES member(memberid)
);

CREATE TABLE paper (
	paperid INTEGER PRIMARY KEY AUTO_INCREMENT,
	content NVARCHAR(50) NOT NULL,
	regdate TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
	sendmemseq INTEGER NOT NULL,
	receivememseq INTEGER NOT NULL,
	kind INTEGER NOT NULL,
	FOREIGN KEY(sendmemseq) REFERENCES member(memberid),
	FOREIGN KEY(receivememseq) REFERENCES member(memberid)
);

CREATE TABLE acceptance (
	acceptanceid INTEGER PRIMARY KEY AUTO_INCREMENT,
	paperseq INTEGER NOT NULL,
	accept INTEGER NULL,
	FOREIGN KEY(paperseq) REFERENCES paper(paperid)
);

CREATE TABLE board (
	boardid INTEGER PRIMARY KEY AUTO_INCREMENT,
	post_type NVARCHAR(10) NOT NULL,
	memseq INTEGER NOT NULL,
	subject NVARCHAR(200) NOT NULL,
	content NVARCHAR(1000) NOT NULL,

	recomCount INTEGER NOT NULL,
	imgfilename VARCHAR(100) NULL,
	imgorgfilename VARCHAR(100) NULL,

CREATE TABLE boardcomment (
	commentid INTEGER PRIMARY KEY AUTO_INCREMENT,
	boardseq INTEGER NOT NULL,
	memseq INTEGER NOT NULL,
	regdate TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
	content NVARCHAR(100) NULL,
	FOREIGN KEY(boardseq) REFERENCES board(boardid),
	FOREIGN KEY(memseq) REFERENCES member(memberid)

CREATE TABLE commentlabor (

CREATE TABLE labor (
	laborid INTEGER PRIMARY KEY AUTO_INCREMENT,

-- View
CREATE VIEW projectMember
AS
SELECT t.teamId, p.projectId, p.name AS projectname, p.thumnail, p.startday AS projectStart, p.endday AS projectEnd, m.memberId, m.name AS membername, m.email, m.profile FROM  member m INNER JOIN team t ON m.memberId = t.memseq INNER JOIN project p ON t.projectseq = p.projectid ORDER BY p.projectid ASC;

CREATE VIEW friendsList
AS
SELECT m1.memberId AS memberId, m1.name AS membername, m1.profile AS profile, m2.memberId AS friendId, m2.name AS friendname, m2.tel, m2.email FROM friend f INNER JOIN member m1 ON m1.memberId = f.memseq INNER JOIN member m2 ON m2.memberId = f.friendseq;

SELECT * FROM friendsList;

CREATE VIEW paperList
AS
SELECT p.paperid, p.kind, p.projectid, m2.memberId AS toId, m1.memberId AS fromId, p.content, p.regdate, m1.name AS fromname, m2.name AS toname FROM member m1 INNER JOIN paper p ON p.sendmemseq = m1.memberId INNER JOIN member m2 ON p.receivememseq = m2.memberId;

CREATE VIEW paperPageBar
AS
SELECT COUNT(*) AS total, CEIL(COUNT(*) / 10) AS pages FROM paper;

CREATE VIEW myproject
AS
SELECT p.projectId, p.name AS projectName, p.startday AS projectStart, p.endday AS projectEnd, p.thumnail, p.twoseq, p.skin, bp.name AS bpName, bp.startday AS bpStart, bp.endday AS bpEnd, explanation AS bpExplan, bp.state AS bpStatus, b.subject AS businessSubject, b.content AS businessContent, b.importance, b.startday AS businessStart, b.endday AS businessEnd, tb.teamseq FROM project p INNER JOIN bp ON p.projectId = bp.projectSeq LEFT OUTER JOIN business b ON bp.bpid = b.bpseq LEFT OUTER JOIN teambusiness t ON b.businessId = t.businessSeq LEFT OUTER JOIN teambp tb ON t.teambpseq = tb.teambpid;

COMMIT;

create view VteamMainList
as select b.startday as startday, b.endday as endday, b.subject as subject, b.bpseq as bpseq, b.businessid as businessid, m.name as name,  t.projectseq as projectseq, b.importance as importance, b.content as content, t.teamid as teamid from business b 
	inner join teambusiness tb on b.businessid = tb.businessseq
		inner join teambp tp on tb.teambpseq = tp.teambpid
			inner join team t on tp.teamseq = t.teamid
				inner join member m on t.memseq = m.memberId;

create view VteamBusinessLog
as select l.businesslogid as businesslogid, l.businessseq as businessseq, s.status as status, 
	l.editdate as editdate 
	from businesslog l 
		inner join businesstatus s on l.businesstatussseq = s.businesstatusid;

create view VteamBusinessList_addLog
as select vm.startday as startday, vm.endday as endday, vm.subject as subject, vm.bpseq as bpseq, 
	vm.businessid as businessid, vm.name as name, vm.projectseq as projectseq, 
	vl.businesslogid as businesslogid, vl.businessseq as businessseq, vl.status as status, 
	vl.editdate as editdate, vm.importance as importance, 
	vm.content as content
from VteamMainList vm left outer join VteamBusinessLog vl on vm.businessid = vl.businessseq;
